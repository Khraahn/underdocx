  - {$For @value:"entities", $as:"subEntity"}
       {$For $value:"subEntity", $as:"x"} ${$x} ${EndIf}
    {$EndFor}
  - {$For @value:"entities", @as:"subEntity"}
       {$For @value:"subEntity", @as:"x"} ${@x} ${EndIf}
    {$EndFor}
  - {$For @value:"entities"}
      ${@}
    {$EndFor}
  - Verboten:
      - {$For $value:"entities"} ...
      - {$For $value:"entities", @as:"x"} ...
